class UnionFind{
public:
    vector<int> root;
    vector<int> rank;
    UnionFind(int n){
        root.resize(n);
        rank.resize(n);
        for(int i = 0; i < n;++i){
            root[i] = i;
            rank[i] = 1;
        }
    }
    int find(int x){
        if(root[x] == x)return x;
        root[x] = find(root[x]);
        return root[x];
    }
    bool merge(int x,int y){
        int fx = find(x),fy = find(y);
        if(fx == fy)return false;
        if(rank[fx] < rank[fy])
            swap(fx,fy);
        root[fy] = fx;
        if(rank[fx] == rank[fy])
            ++rank[fx];
        return true;
    }
    bool isConnected(int x,int y){
        int fx = find(x),fy = find(y);
        return fx == fy;
    }
    bool disConnect(int x){
        root[x] = x;
        return true;
    }
};
class Solution {
public:
    vector<int> findAllPeople(int n, vector<vector<int>>& meetings, int firstPerson) {
        sort(meetings.begin(),meetings.end(),[](auto &a,auto &b){
           return a[2] < b[2]; 
        });
        UnionFind q(n);
        q.merge(0,firstPerson);
        int m = meetings.size();
        for(int i = 0; i < m;++i){
            //同一时间的建立连接
            int j = i+1;
            while(j < m && meetings[j][2] == meetings[i][2])++j;
            //开始合并
            for(int k = i; k < j;++k){
                q.merge(meetings[k][0],meetings[k][1]);
            }
            //然后判断是不是跟已知的专家一块的，
            for(int k = i; k < j;++k){
                if(!q.isConnected(meetings[k][0],0)){
                    q.disConnect(meetings[k][0]);
                    q.disConnect(meetings[k][1]);
                }
            }
            i = j-1;
        }
        vector<int> res;
        for(int i = 0; i < n;++i){
            if(q.isConnected(0,i))res.push_back(i);
        }
        return res;
    }
};
